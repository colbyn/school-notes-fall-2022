Pythagorean Identities

\begin{equation*}\begin{split}\cos^2\left(\theta\right) + \sin^2\left(\theta\right) = 1\end{split}\end{equation*}
\begin{equation*}\begin{split}\sec^2\left(\theta\right) - \tan^2\left(\theta\right) &= 1 \\ \sec^2\left(\theta\right) &= 1 + \tan^2\left(\theta\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\csc^2\left(\theta\right) - \cot^2\left(\theta\right) &= 1 \\ \csc^2\left(\theta\right) &= 1 + \cot^2\left(\theta\right)\end{split}\end{equation*}

Sum and Difference Identities

\begin{equation*}\begin{split}\cos\left(\alpha - \beta\right) &= \cos\left(\alpha\right) \cdot \cos\left(\beta\right) + \sin\left(\alpha\right) \cdot \sin\left(\beta\right) \\ \cos\left(\alpha + \beta\right) &= \cos\left(\alpha\right) \cdot \cos\left(\beta\right) - \sin\left(\alpha\right) \cdot \sin\left(\beta\right) \\ &\\ \sin\left(\alpha - \beta\right) &= \sin\left(\alpha\right) \cdot \cos\left(\beta\right) - \cos\left(\alpha\right) \cdot \sin\left(\beta\right) \\ \sin\left(\alpha + \beta\right) &= \sin\left(\alpha\right) \cdot \cos\left(\beta\right) + \cos\left(\alpha\right) \cdot \sin\left(\beta\right) \\ &\\ \tan\left(\alpha + \beta\right) &= \frac{\tan\left(\alpha\right) + \tan\left(\beta\right)}{1 - \tan\left(\alpha\right) \cdot \tan\left(\beta\right)} \\ \tan\left(\alpha - \beta\right) &= \frac{\tan\left(\alpha\right) - \tan\left(\beta\right)}{1 + \tan\left(\alpha\right) \cdot \tan\left(\beta\right)}\end{split}\end{equation*}

Cofunction Identities

\begin{equation*}\begin{split}\cos\left(\theta\right) = \sin\left(\frac{1}{4}\tau - \theta\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\tan\left(\theta\right) = \cot\left(\frac{1}{4}\tau - \theta\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\cot\left(\theta\right) = \tan\left(\frac{1}{4}\tau - \theta\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\csc\left(\theta\right) = \sec\left(\frac{1}{4}\tau - \theta\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\sec\left(\theta\right) = \csc\left(\frac{1}{4}\tau - \theta\right)\end{split}\end{equation*}

Ratio Identities

\begin{equation*}\begin{split}\tan\left(90^\circ - x\right) & = \frac{\sin\left(90^\circ - x\right)}{\cos\left(90^\circ - x\right)} = \frac{\cos\left(x\right)}{\sin\left(x\right)} = \cot\left(x\right) \\ \\ \cot\left(90^\circ - x\right) & = \frac{\cos\left(90^\circ - x\right)}{\sin\left(90^\circ - x\right)} = \frac{\sin\left(x\right)}{\cos\left(x\right)} = \tan\left(x\right) \\\end{split}\end{equation*}

Double-Angle Identities

\begin{equation*}\begin{split}\sin\left(2\alpha\right) &= 2\sin\left(\alpha\right)\cos\left(\alpha\right) \\ \cos\left(2\alpha\right) &= \cos^2\left(\alpha\right) - \sin^2\left(\alpha\right) \\ &= 1 - 2\sin^2\left(\alpha\right) \\ &= 2\cos^2\left(\alpha\right) - 1 \\ &\\ \tan\left(2\alpha\right) &= \frac{2\tan\left(\alpha\right)}{1 - \tan^2\left(\alpha\right)}\end{split}\end{equation*}

Half-Angle Identities

\begin{equation*}\begin{split}\sin \frac{\alpha}{2} &= \pm \sqrt{\frac{1 - \cos\left(\alpha\right)}{2}} \\ \cos \frac{\alpha}{2} &= \pm \sqrt{\frac{1 + \cos\left(\alpha\right)}{2}} \\ \tan \frac{\alpha}{2} &= \pm \sqrt{\frac{1 - \cos\left(\alpha\right)}{1 + \cos\left(\alpha\right)}} \\ &= \frac{sin\left(\alpha\right)}{1 + \cos\left(\alpha\right)} \\ &= \frac{1 - \cos\left(\alpha\right)}{sin\left(\alpha\right)}\end{split}\end{equation*}

Power-Reducing Identities

\begin{equation*}\begin{split}\sin^2\left(\alpha\right) &= \frac{1 - \cos\left(2\alpha\right)}{2} \\ &\\ \cos^2\left(\alpha\right) &= \frac{1 + \cos\left(2\alpha\right)}{2} \\ &\\ \tan^2\left(\alpha\right) &= \frac{1 - \cos\left(2\alpha\right)}{1 + \cos\left(2\alpha\right)}\end{split}\end{equation*}
\begin{equation*}\begin{split}\sin\alpha\cdot\cos\alpha &= \frac{1}{2}\sin\left(2\alpha\right)\end{split}\end{equation*}

Product-to-Sum Identities

\begin{equation*}\begin{split}\sin\left(\alpha\right) \cdot \cos\left(\beta\right) &= \frac{1}{2} \left[\sin\left(\alpha + \beta\right) + \sin\left(\alpha - \beta\right)\right] \\ \cos\left(\alpha\right) \cdot \sin\left(\beta\right) &= \frac{1}{2} \left[\sin\left(\alpha + \beta\right) - \sin\left(\alpha - \beta\right)\right] \\ \cos\left(\alpha\right) \cdot \cos\left(\beta\right) &= \frac{1}{2} \left[\cos\left(\alpha + \beta\right) + \cos\left(\alpha - \beta\right)\right] \\ \sin\left(\alpha\right) \cdot \sin\left(\beta\right) &= \frac{1}{2} \left[\cos\left(\alpha - \beta\right) - \cos\left(\alpha + \beta\right)\right]\end{split}\end{equation*}

Sum-to-Product-Identities

\begin{equation*}\begin{split}\sin\left(x\right) + \sin\left(y\right) &= 2 \cdot \sin\left(\frac{x + y}{2}\right) \cdot \cos\left(\frac{x - y}{2}\right) \\ \cos\left(x\right) + \cos\left(y\right) &= 2 \cdot \cos\left(\frac{x + y}{2}\right) \cdot \cos\left(\frac{x - y}{2}\right) \\ \sin\left(x\right) - \sin\left(y\right) &= 2 \cdot \cos\left(\frac{x + y}{2}\right) \cdot \sin\left(\frac{x - y}{2}\right) \\ \cos\left(x\right) - \cos\left(y\right) &= -2 \sin \cos\left(\frac{x + y}{2}\right) \cdot \sin\left(\frac{x - y}{2}\right)\end{split}\end{equation*}

Euler's Formula

\begin{equation*}\begin{split}e^{i x} &= \cos\left(x\right) + \mathrm{i}\sin\left(x\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\mathrm{e} &= \sum_{n = 0}^\infty \frac{1}{n!} \\ &= \lim_{n\to\infty}\left(1 + \frac{1}{n}\right)^n \\ &= \lim_{t \to 0}\left(1 + t\right)^{\frac{1}{t}}\end{split}\end{equation*}
\begin{equation*}\begin{split}\mathrm{e}^x &= 1 + \frac{x}{1!} + \frac{x^2}{2!} + \frac{x^3}{3!} + \cdots \\ &= \sum_{n = 0}^\infty \frac{x^n}{n!} \\ &= \lim_{n\to\infty}\left(1 + \frac{x}{n}\right)^n\end{split}\end{equation*}
