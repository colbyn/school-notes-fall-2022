Pythagorean Identities

\begin{equation*}\begin{split}\cos^2\left(\theta\right) + \sin^2\left(\theta\right) = 1\end{split}\end{equation*}
\begin{equation*}\begin{split}\sec^2\left(\theta\right) - \tan^2\left(\theta\right) &= 1 \\ \sec^2\left(\theta\right) &= 1 + \tan^2\left(\theta\right)\end{split}\end{equation*}
\begin{equation*}\begin{split}\csc^2\left(\theta\right) - \cot^2\left(\theta\right) &= 1 \\ \csc^2\left(\theta\right) &= 1 + \cot^2\left(\theta\right)\end{split}\end{equation*}

Sum and Difference Identities

\begin{equation*}\begin{split}\cos\left(\alpha - \beta\right) &= \cos\left(\alpha\right) \cdot \cos\left(\beta\right) + \sin\left(\alpha\right) \cdot \sin\left(\beta\right) \\ \cos\left(\alpha + \beta\right) &= \cos\left(\alpha\right) \cdot \cos\left(\beta\right) - \sin\left(\alpha\right) \cdot \sin\left(\beta\right) \\ &\\ \sin\left(\alpha - \beta\right) &= \sin\left(\alpha\right) \cdot \cos\left(\beta\right) - \cos\left(\alpha\right) \cdot \sin\left(\beta\right) \\ \sin\left(\alpha + \beta\right) &= \sin\left(\alpha\right) \cdot \cos\left(\beta\right) + \cos\left(\alpha\right) \cdot \sin\left(\beta\right) \\ &\\ \tan\left(\alpha + \beta\right) &= \frac{\tan\left(\alpha\right) + \tan\left(\beta\right)}{1 - \tan\left(\alpha\right) \cdot \tan\left(\beta\right)} \\ \tan\left(\alpha - \beta\right) &= \frac{\tan\left(\alpha\right) - \tan\left(\beta\right)}{1 + \tan\left(\alpha\right) \cdot \tan\left(\beta\right)}\end{split}\end{equation*}

Cofunction Identities

Ratio Identities

\begin{equation*}\begin{split}\tan\left(90^\circ - x\right) & = \frac{\sin\left(90^\circ - x\right)}{\cos\left(90^\circ - x\right)} = \frac{\cos\left(x\right)}{\sin\left(x\right)} = \cot\left(x\right) \\ \\ \cot\left(90^\circ - x\right) & = \frac{\cos\left(90^\circ - x\right)}{\sin\left(90^\circ - x\right)} = \frac{\sin\left(x\right)}{\cos\left(x\right)} = \tan\left(x\right) \\\end{split}\end{equation*}
